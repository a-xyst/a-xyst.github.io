#### 最大公约数

读入两个数a,b
输出a和b的最大公约数

```c++
long long gcd(long long a,long long b)
{
long long t;
if(a<b) {t=a;a=b;b=t;}
while(b>0)
{  
    t=a%b;  
    a=b;  
    b=t;  
}
return a;
}
int main()
{
    long long x,y;
    scanf("%I64d%I64d",&x,&y);
    long long a=gcd(x,y);
    printf("%I64d",a);
    return 0;
}
```

#### 区间和

```c++
int sum[1000];
int a[1000];
int main()
{
    int x,y;
```

```c++
int lcm(int a,int b)
```

```c++
    if(n==0||n==1) return 1;
```

```c++
#include<iostream>
using namespace std;
int fib(int n)
{
    if(n==0||n==1) return 1;
    else
    {
    int a=1,b=1,t;
    for(int i=3;i<=n+1;i++)
    {
    t=a+b;
    a=b;
    b=t;
    }
    return t;
}
}
int main()
{
    int n;
    scanf("%d",&n);
    printf("%d\n",fib(n));
    return 0;
}
```
