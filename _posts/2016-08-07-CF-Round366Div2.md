705B Spider Man

题意：两个人玩游戏，对于每个输入可以任意把它分成两堆（新的堆如果大于1可以继续分），如果某个人不能进行操作他就输了，问每轮输入后谁能赢。

这题主要难在读题，规律还是很好找的。因为数据范围比较大，所以一开始就觉得应该有直接判别法。

```c
#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<math.h>
#include<string.h>
#include<queue>
#include<vector>
#include<map>
#include<iostream>
#include<string>
#define LL long long
using namespace std;
int main()
{
//freopen("test.txt","r",stdin);
    int n,now=1,tmp;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%d",&tmp);
        if(now==1)
        {
            if(tmp%2==1)
            {
                printf("2\n");
                now=1;
            }
            if(tmp%2==0)
            {
                printf("1\n");
                now=2;
            }
            continue;
        }
        if(now==2)
        {
            if(tmp%2==1)
            {
                printf("1\n");
                now=2;
            }
            if(tmp%2==0)
            {
                printf("2\n");
                now=1;
            }
        }
    }
}
```
